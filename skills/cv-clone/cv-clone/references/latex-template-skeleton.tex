% This is the working scaffold the cv-clone agent extends per sample.
% All user-fillable slots are exposed as `\newcommand` with defaults.
%
% Placeholders (override via `\renewcommand{...}` at the top):
%   \cvname        Name
%   \cvlocation    City
%   \cvemail       Email
%   \cvphone       Phone
%   \cvwebsite     Personal site / GitHub
%   \cvobjective   One-paragraph career goal
%   \cvskillrows   Skill items (itemize)
%   \cvexperience  Work experience entries
%   \cvprojects    Project entries
%   \cveducation   Education entries
%   \cvothers      Other entries
%   \cvphoto{path} Optional photo (only if the sample has a photo slot)
%
% Compile:  tectonic template.tex   (or xelatex template.tex)
% Output:   template.pdf + template_preview-1.png (via pdftoppm -r 150)
% =====================================================================

\documentclass[11pt,a4paper]{article}

% ---------- Default placeholder values (override with \renewcommand) ----------
\newcommand{\cvname}{NAME}
\newcommand{\cvlocation}{CITY}
\newcommand{\cvemail}{name@example.com}
\newcommand{\cvphone}{+1-000-000-0000}
\newcommand{\cvwebsite}{github.com/handle}
\newcommand{\cvobjective}{Career goal: a one-paragraph description of the role you are targeting.}

% ---------- Colors ----------
\definecolor{cvrule}{HTML}{1A1A1A}

% ---------- Global typography ----------
\pagestyle{empty}
\setlength{\parindent}{0pt}

\setlist[itemize]{
  leftmargin=1.6em,
  labelsep=0.6em,
  itemsep=1pt,
  parsep=0pt,
  topsep=2pt,
  label=\textbullet
}

% ---------- Section title: bold + thin horizontal rule ----------
\newcommand{\cvsection}[1]{%
  \par\vspace{6pt}%
  {\noindent\bfseries\fontsize{12.5pt}{15pt}\selectfont #1}%
  \par\vspace{1pt}%
  {\color{cvrule}\hrule height 1.1pt}%
  \vspace{4pt}%
  \noindent\ignorespaces
}

% ---------- Entry first line: title-left, date-right ----------
\newcommand{\cventry}[2]{%
  \par\noindent
  {\bfseries #1}\hfill{\normalfont\color{cvtext}#2}%
  \par\vspace{2pt}%
}

% ---------- Skill item: bold label + content ----------
\newcommand{\cvskill}[2]{%
  \item \textbf{#1}: #2
}
